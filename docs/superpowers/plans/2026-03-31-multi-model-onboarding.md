```python
# ─── Catalog cache (thread-safe, per model) ────────────────────────────────
_catalog_cache: dict[int, list[dict]] = {}
_catalog_cache_lock = threading.Lock()


def get_catalog_groups(session: Session, bike_model_id: int) -> list[dict]:
    """Charge le catalogue d'un modele depuis la DB (avec cache thread-safe)."""
    with _catalog_cache_lock:
        if bike_model_id in _catalog_cache:
            return _catalog_cache[bike_model_id]

    groups = session.exec(
        select(AccessoryCatalogGroup)
        .where(AccessoryCatalogGroup.bike_model_id == bike_model_id)
        .options(selectinload(AccessoryCatalogGroup.variants))
        .order_by(AccessoryCatalogGroup.category)
    ).all()

    result = []
    for g in groups:
        result.append({
            "id": g.id,
            "group_key": g.group_key,
            "bike_model_id": g.bike_model_id,
            "name": g.name,
            "category": g.category,
            "expressions": g.expressions or [],
            "default_price": g.default_price,
            "last_match_count": g.last_match_count,
            "created_at": g.created_at,
            "updated_at": g.updated_at,
            "variants": [
                {
                    "id": v.id,
                    "group_id": v.group_id,
                    "name": v.name,
                    "qualifiers": v.qualifiers or [],
                    "brands": v.brands or [],
                    "product_aliases": v.product_aliases or [],
                    "optional_words": v.optional_words or [],
                    "regex_override": v.regex_override,
                    "estimated_new_price": v.estimated_new_price,
                    "sort_order": v.sort_order,
                    "sort_order_manual": v.sort_order_manual,
                    "notes": v.notes,
                    "created_at": v.created_at,
                    "updated_at": v.updated_at,
                }
                for v in sorted(g.variants, key=lambda v: v.sort_order)
            ],
        })

    with _catalog_cache_lock:
        _catalog_cache[bike_model_id] = result
    return result


def invalidate_catalog_cache(bike_model_id: int | None = None) -> None:
    """Invalide le cache catalogue. Si bike_model_id est None, invalide tout."""
    with _catalog_cache_lock:
        if bike_model_id is None:
            _catalog_cache.clear()
        else:
            _catalog_cache.pop(bike_model_id, None)
```

```python
def create_bike_model(session: Session, data: dict) -> BikeModel:
    """Cree un modele de moto avec config par defaut et search config."""
    import re
    from .catalog import normalize_text

    # Generate slug if not provided
    slug = data.get("slug")
    if not slug:
        slug = normalize_text(f"{data['brand']} {data['name']}").strip()
        slug = re.sub(r"[^a-z0-9]+", "-", slug).strip("-")

    # Check slug format and uniqueness
    if not re.match(r"^[a-z0-9-]+$", slug):
        raise ValueError(f"Slug invalide: {slug}")
    existing = session.exec(select(BikeModel).where(BikeModel.slug == slug)).first()
    if existing:
        raise ValueError(f"slug_conflict:{slug}")

    now = datetime.now().isoformat()

    model = BikeModel(
        slug=slug,
        brand=data["brand"],
        name=data["name"],
        engine_cc=data["engine_cc"],
        image_url=data.get("image_url"),
        active=True,
        created_at=now,
    )
    session.add(model)
    session.flush()  # Get model.id

    # Default config (sensible defaults)
    config = BikeModelConfig(
        bike_model_id=model.id,
        warranty_years=3,
        warranty_value_per_year=200,
        mechanical_wear_per_km=0.03,
        condition_risk_per_km=0.04,
        short_term_km_threshold=3000,
    )
    session.add(config)

    # Search config
    sc = data.get("search_config", {})
    search_config = BikeSearchConfig(
        bike_model_id=model.id,
        keyword=sc["keyword"],
        min_cc=sc.get("min_cc", data["engine_cc"]),
        max_cc=sc.get("max_cc", data["engine_cc"] + 5),
        locations=sc.get("locations"),
        owner_type=sc.get("owner_type"),
        price_min=sc.get("price_min"),
        price_max=sc.get("price_max"),
        sort=sc.get("sort"),
        search_in_title_only=sc.get("search_in_title_only", False),
    )
    session.add(search_config)

    session.commit()
    session.refresh(model)
    return model
```

```python
def delete_bike_model(session: Session, slug: str) -> None:
    """Supprime un modele si aucune annonce n'existe."""
    model = session.exec(select(BikeModel).where(BikeModel.slug == slug)).first()
    if not model:
        raise ValueError(f"Modele '{slug}' non trouve")

    ad_count = session.exec(
        select(func.count()).select_from(Ad).where(Ad.bike_model_id == model.id)
    ).one()
    if ad_count > 0:
        raise ValueError(f"cannot_delete_has_ads:{ad_count}")

    # Delete related data (config, search configs, catalog groups cascade)
    session.exec(delete(BikeSearchConfig).where(BikeSearchConfig.bike_model_id == model.id))
    session.exec(delete(BikeModelConfig).where(BikeModelConfig.bike_model_id == model.id))
    session.exec(delete(AccessoryCatalogGroup).where(AccessoryCatalogGroup.bike_model_id == model.id))
    session.exec(delete(BikeVariant).where(BikeVariant.bike_model_id == model.id))
    session.exec(delete(BikeConsumable).where(BikeConsumable.bike_model_id == model.id))
    session.exec(delete(BikeAccessoryPattern).where(BikeAccessoryPattern.bike_model_id == model.id))
    session.exec(delete(BikeVariantPattern).where(BikeVariantPattern.bike_model_id == model.id))
    session.exec(delete(BikeExclusionPattern).where(BikeExclusionPattern.bike_model_id == model.id))
    session.exec(delete(BikeNewListingPattern).where(BikeNewListingPattern.bike_model_id == model.id))
    session.delete(model)
    session.commit()
    invalidate_catalog_cache(model.id)
```
