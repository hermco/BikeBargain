> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace binary `sold` (0/1) with tri-state `listing_status` (online/paused/sold), add status history tracking, and provide quick + full check modes in the UI.

**Architecture:** New `listing_status` field on `Ad` replaces `sold`. New `AdStatusHistory` table logs every transition. LBC service returns raw status; API maps it to app states. Frontend shows 3 visual states and offers 2 check buttons.

**Tech Stack:** Python/FastAPI/SQLModel/Alembic (backend), React 19/TypeScript/TanStack Query (frontend), PostgreSQL

**Important naming note:** `Ad.status` (line 204 of `src/models.py`) already exists and stores the raw LBC status from extraction. The new field is named `listing_status` to avoid collision.

def _map_lbc_status(reason: str) -> str:
    """Map LBC check result reason to app listing_status."""
    if reason.startswith("status="):
        lbc_status = reason.split("=", 1)[1]
        if lbc_status == "paused":
            return "paused"
    # "inaccessible", "status=deleted", or any other non-active → sold
    return "sold"


def _log_status_change(session: Session, ad: Ad, new_status: str, reason: str | None = None):
    """Log a status transition to ad_status_history if status actually changed."""
    old_status = ad.listing_status
    if old_status == new_status:
        return False
    session.add(AdStatusHistory(
        ad_id=ad.id,
        old_status=old_status,
        new_status=new_status,
        reason=reason,
    ))
    ad.listing_status = new_status
    ad.updated_at = datetime.now().isoformat()
    return True
```

@app.get("/api/bike-models/{slug}/ads/{ad_id}/status-history")
def get_status_history_scoped(slug: str, ad_id: int, session: Session = Depends(get_session)):
    model = resolve_bike_model(slug, session)
    ad = session.get(Ad, ad_id)
    if not ad or ad.bike_model_id != model.id:
        raise HTTPException(status_code=404, detail="Annonce non trouvée")
    entries = session.exec(
        select(AdStatusHistory)
        .where(AdStatusHistory.ad_id == ad_id)
        .order_by(AdStatusHistory.changed_at.desc())
    ).all()
    return {
        "ad_id": ad_id,
        "current_status": ad.listing_status,
        "history": [
            {
                "old_status": e.old_status,
                "new_status": e.new_status,
                "reason": e.reason,
                "changed_at": e.changed_at,
            }
            for e in entries
        ],
    }
```
